# Simple Node structure


Simple Node structure for CC150 chap2 problems





{% highlight ruby %}

import java.util.*;
public class Node<E> {
     E val;
     Node<E> prev;
     Node<E> next;
     
     public Node(E val, Node<E> prev, Node<E> next){
    	 this.val = val;
    	 this.prev = prev;
    	 this.next = next;
     }
}
{% endhighlight %}
